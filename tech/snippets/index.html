    change all file with JPG extension to jpg (lower-case)
    from pathlib import Path
import os

for r, d, f in os.walk('/path/to'):
    for x in f:
        if '.JPG' in x:
            p = Path(r, x)
            p.rename(Path(r, x.replace('.JPG', 'jpg')))
